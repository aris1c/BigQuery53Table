SELECT 
    A.TGL,
    O.OUTLET_PBI,
    A.J_PATIENT + A.R_PATIENT AS PATIENT,
	A.J_TEST + A.R_TEST AS TEST,
    A.BRUTO,
    A.DISKON,
    A.NETTO,
    A.RETUR,
    A.NETTO - A.RETUR  AS TOTAL
FROM 
    SALES1_ARIS A
LEFT JOIN 
    OUTLET_PBI_DET O ON A.OUTLET_ID = O.OUTLET_ID
WHERE 
    TRUNC(A.TGL) >= TRUNC(SYSDATE-120);
